CREATE procedure [dbo].[SP_TimSV]
@MASV NCHAR(10), @PASSWORD NVARCHAR(40)
AS 
If  exists(select MASV from  DBO.SINHVIEN where MASV =@MASV AND PASSWORD = @PASSWORD)
BEGIN
	RETURN 1
END
ELSE
	IF EXISTS(SELECT MASV FROM LINK0.QLDSV.DBO.SINHVIEN WHERE MASV = @MASV AND PASSWORD = @PASSWORD)
	BEGIN
		RETURN 2
	END
ELSE 
	RETURN 0
GO
